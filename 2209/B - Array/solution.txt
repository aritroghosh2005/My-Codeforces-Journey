import sys
input = lambda: sys.stdin.readline().rstrip()
print = lambda *args: sys.stdout.write(' '.join(map(str, args)) + '
')
 
def solve():
    n = int(input())
    a = list(map(int, input().split()))
    answers = []
 
    for i in range(n):
        less = sum(1 for j in range(i+1, n) if a[j] > a[i])
        greater = sum(1 for j in range(i+1, n) if a[j] < a[i])
        answers.append(max(less, greater))
    print(*answers)
 
t = int(input())
for _ in range(t):
    solve()
