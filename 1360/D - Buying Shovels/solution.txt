import sys
input = lambda : sys.stdin.readline().rstrip()
 
from math import isqrt
 
def solve():
    n, k = map(int, input().split())
    needed = n
 
    for i in range(1, isqrt(n)+1):
        if n%i == 0:
            if i <= k: needed = min(n//i, needed)
            if n//i <= k: needed = min(i, needed)
    print(needed)
 
t = int(input())
for _ in range(t): solve()
