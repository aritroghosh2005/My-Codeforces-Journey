import sys
input = lambda : sys.stdin.readline().rstrip()
 
from math import isqrt
 
def solve():
    n, k = map(int, input().split())
 
    factors = []
    for i in range(1, isqrt(n)+1):
        if n%i==0: 
            factors.append(i)
            if i*i != n: factors.append(n//i)
    factors.sort(reverse=True)
    
    for f in factors:
        if f <= k:
            print(n//f); break
 
t = int(input())
for _ in range(t): solve()
