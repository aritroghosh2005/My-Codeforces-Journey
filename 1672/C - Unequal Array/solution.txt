import sys
input = lambda : sys.stdin.readline().rstrip()
 
def solve():
    n = int(input())
    a = list(map(int, input().split()))
 
    start = end = -1
    for i in range(n-1):
        if a[i] == a[i+1]:
            if start == -1: start = i
            end = i

    elif start == end - 1: op = 1
