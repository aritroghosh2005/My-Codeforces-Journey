import sys
input = lambda : sys.stdin.readline().rstrip()
 
def solve():
    n = int(input())
    b = list(map(int, input().split()))
 
    a = [1]
    nextNumber = 2
 
    for i in range(1, n):
        j = b[i] - b[i-1] - 1
        if i == j: 
            a.append(nextNumber)
            nextNumber += 1
        else: a.append(a[i-j-1])
    print(*a)    
 
t = int(input())
for _ in range(t): solve()
