import sys
input = lambda : sys.stdin.readline().rstrip()
 
def solve():
    n = input()
 
    for i in range(0, 1000, 8):
        target = str(i)
        nIter = iter(n)
 
        if all(dig in nIter for dig in target):
            print("YES")
            print(target)
            return
    print("NO")
 
t = 1
for _ in range(t): solve()
