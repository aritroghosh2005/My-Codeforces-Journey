import sys
input = lambda: sys.stdin.readline().rstrip()
print = lambda *args: sys.stdout.write(' '.join(map(str, args)) + '
')
 
def solve():
    n, x, t = map(int, input().split())
    c = t // x
 
    if n <= c: dis = n*(n-1) // 2
    else: dis = c*(c-1) // 2 + (n-c)*c
    print(dis)
 
t = int(input())
for _ in range(t):
    solve()
