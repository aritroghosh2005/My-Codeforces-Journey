import sys
input = lambda : sys.stdin.readline().rstrip()
 
def solve():
    n = int(input())
    a = list(map(int, input().split()))
 
    a.sort()
    coins = 0
 
    for i in range(1, n):
        if a[i] <= a[i-1]:
            target = a[i-1] + 1
            coins += (target-a[i])
            a[i] = target
    print(coins)
 
t = 1
for _ in range(t): solve()
