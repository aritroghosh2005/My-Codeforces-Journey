import sys
input = lambda : sys.stdin.readline().rstrip()
 
def solve():
    n = int(input())
    a = list(map(int, input().split()))
 
    runningSum = 0
    for i in range(1, n+1):
        runningSum += a[i-1]
        minRequired = i*(i+1) // 2
        if runningSum < minRequired: print("NO"); return
    print("YES")
 
t = int(input())
for _ in range(t): solve()
