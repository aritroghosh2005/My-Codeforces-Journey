import sys
input = lambda : sys.stdin.readline().rstrip()
 
from math import gcd
 
def solve():
    n, x, y = map(int, input().split())
    p = list(map(int, input().split()))
    g = gcd(x,y)
    
    for i in range(n):
        present, target = i, p[i] - 1
        diff = abs(target - present)
 
        if diff%g != 0: print("NO"); return
    print("YES") 
 
t = int(input())
for _ in range(t): solve()
