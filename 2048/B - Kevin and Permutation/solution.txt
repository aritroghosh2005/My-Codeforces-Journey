import sys
input = lambda : sys.stdin.readline().rstrip()
 
import gc
gc.disable()
 
def solve():
    n, k = map(int, input().split())
    g = n//k
 
    anchors = iter(range(1, g+1))
    bulk = iter(range(g+1, n+1))
 
    perm = [0]*n
    for i in range(1, n+1):
        if i%k==0: perm[i-1] = next(anchors)
        else: perm[i-1] = next(bulk)
    sys.stdout.write(' '.join(map(str, perm)) + '
')
 
t = int(input())
for _ in range(t): solve()
