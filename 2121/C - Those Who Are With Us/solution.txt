import sys
input = lambda : sys.stdin.readline().rstrip()
 
from itertools import product
 
def solve():
    n, m = map(int, input().split())
    matrix = []
 
    for u in range(n):
        row = list(map(int, input().split()))
        matrix.append(row)
    
    val = max(max(row) for row in matrix)
    rowCounts, columnCounts = [0]*n, [0]*m
    freq = 0
    
    for i, j in product(range(n), range(m)):
        if matrix[i][j] == val:
            rowCounts[i] += 1
            columnCounts[j] += 1
            freq += 1
    
    for i, j in product(range(n), range(m)):
        covered = rowCounts[i] + columnCounts[j]
        if matrix[i][j] == val: covered -= 1
 
        if covered == freq: print(val-1); return
    print(val)
 
t = int(input())
for _ in range(t): solve()
