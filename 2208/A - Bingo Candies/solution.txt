from collections import Counter
 
def solve():
    n = int(input())
    
    matrix = []
    for i in range(n):
        row = list(map(int, input().split()))
        matrix.extend(row)
    
    freq = Counter(matrix)
    
    if max(freq.values()) > n*(n-1): print("NO")
    else: print("YES")
 
 
t = int(input())
for _ in range(t):
    solve()
