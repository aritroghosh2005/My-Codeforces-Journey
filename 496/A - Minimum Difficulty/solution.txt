import sys
input = lambda : sys.stdin.readline().rstrip()
 
def solve():
    n = int(input())
    a = list(map(int, input().split()))
    
    originalDifficulty = max(a[i+1] - a[i] for i in range(n-1))
    minNewGap = min(a[i+1] - a[i-1] for i in range(1, n-1))
    print(max(originalDifficulty, minNewGap))
 
t = 1
for _ in range(t): solve()
