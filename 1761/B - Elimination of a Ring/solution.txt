import sys
input = lambda: sys.stdin.readline().rstrip()
print = lambda *args: sys.stdout.write(' '.join(map(str, args)) + '
')
 
from functools import reduce
from operator import mul
 
def solve():
    n = int(input())
    a = list(map(int, input().split()))
    counts = {}
 
    for number in a:
        if number not in counts: counts[number] = 1
        else: counts[number] += 1
    
    if len(counts.keys()) > 2: print(n)
    else: print(n//2 + 1)
    
t = int(input())
for _ in range(t):
    solve()
