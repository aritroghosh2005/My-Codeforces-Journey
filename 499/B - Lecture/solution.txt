import sys
input = lambda : sys.stdin.readline().rstrip()
 
def solve():
    n, m = map(int, input().split())
    translator = {}
 
    for __ in range(m):
        a,b = input().split()
        translator[a] = a if len(a) <= len(b) else b
    
    lecture = input().split()
    translation = [translator[word] for word in lecture]
    print(*translation)
 
t = 1
for _ in range(t): solve()
