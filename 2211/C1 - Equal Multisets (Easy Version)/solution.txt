import sys
input = lambda: sys.stdin.readline().rstrip()
print = lambda *args: sys.stdout.write(' '.join(map(str, args)) + '
')
 
def solve():
    n, k = map(int, input().split())
    a = list(map(int, input().split()))
    b = list(map(int, input().split()))
 
    startCommon = n - k
    endCommon = k - 1
    
    seenCommon = set()
    allowedCommon = set()
 
    if startCommon <= endCommon:
        for i in range(startCommon, endCommon + 1):
            allowedCommon.add(a[i])
 
    for i in range(n):
        if b[i] == -1: continue
        if startCommon <= i <= endCommon:
            if b[i] not in allowedCommon or b[i] in seenCommon:
                print("NO")
                return
            seenCommon.add(b[i])
        else:
            if b[i] != a[i]:
                print("NO")
                return
    print("YES")
 
t = int(input())
for _ in range(t):
    solve()
